						& =\overline {d}+A 				\\
		\bar {x}_{i}	& =\sum _{ i=1 }^{ N }{ \frac { { x }_{ i }{ d }_{ i } }{ N } + A  }
	\end{align*}

				\bar{x} & =c\left[ \sum _{ i=1 }^{ N }{ \frac { { f }_{ i }{ u }_{ i } }{ N }  }  \right] +A
	\end{align*}
