	\end{align*}

	\begin{align*}
	Si \quad {x}_{i}	& ={ d }_{ i }+A 				\\
		\bar{x}			& =\overline {d+A} 				\\
						& =\overline {d}+A 				\\
		\bar {x}_{i}	& =\sum _{ i=1 }^{ N }{ \frac { { x }_{ i }{ d }_{ i } }{ N } + A  }
	\end{align*}
\end{minipage} \hfill \begin{minipage}[t]{0.3\textwidth}

	\begin{align*}
	{d}_{i}				& =c{u}_{i}						\\

				\bar{x} & =c\left[ \sum _{ i=1 }^{ N }{ \frac { { f }_{ i }{ u }_{ i } }{ N }  }  \right] +A
	\end{align*}
