    gamma(.5*(\n+1))/(sqrt(\n*pi)*gamma(.5*\n))*((1+x^2/\n)^(-.5*(\n+1)))%
}

            ymin=0,

                smooth,
                no marks,
                usm,
                domain={-6:+6},
                ]{\basefunc};
            \legend{$n = \n$}
        \end{axis}
    \end{tikzpicture}
    
}
